### 生成代码图片
将一段代码生成漂亮的图片 https://carbon.now.sh/  
![image.png](https://upload-images.jianshu.io/upload_images/17817191-ebf6eecc91a4f126.png?imageMogr2/auto-orient/strip%7CimageView2/2/w/1240)
```js
const pluckDeep = key => obj => key.split('.').reduce((accum, key) => accum[key], obj)

const compose = (...fns) => res => fns.reduce((accum, next) => next(accum), res)

const unfold = (f, seed) => {
  const go = (f, seed, acc) => {
    const res = f(seed)
    return res ? go(f, res[1], acc.concat([res[0]])) : acc
  }
  return go(f, seed, [])
}
```
